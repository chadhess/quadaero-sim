prop notation: D×P (e.g. 5×5.5 = 5 in diameter, 5.5 in geometric pitch)
// standard hobby prop label; pitch is advance per revolution in ideal conditions

θ(r) = atan( P_m / (2π · r) ) → θ₀.₇₅ = atan( P_m / (2π · 0.75 · R) )
// geometric pitch → blade pitch angle evaluated at 75% radius (representative station)

chord ≈ 0.12 · D
// empirical rule for typical 2-blade hobby props; adjust for high-solidity or race props

T = ∫ dT = ∫ ½ρV² c [ Cl·cos(φ) − Cd·sin(φ) ] Nb dr
// elemental thrust integrated over blade span

Q = ∫ dQ = ∫ ½ρV² c [ Cd·cos(φ) + Cl·sin(φ) ] r Nb dr
// elemental torque (reaction to rotation)

P = Q · ω
// shaft power from torque and angular velocity

FM = P_ideal / P_actual = T^(3/2) / ( √(2ρA) · P )
// figure of merit — ratio to actuator-disk minimum power; ideal FM = 1

ρ(h) = ρ₀ · (1 − 2.2558×10⁻⁵ · h)^5.256
// ISA standard atmosphere density vs altitude

Cl = 2π·α Cd = Cd₀(Re) + k·α²
// thin airfoil lift + parabolic drag polar (k=0.015)

Cd₀(Re) = max(0.0085, 1.65 · Re⁻⁰·⁴⁰) Re = ρ·V·c / μ
// calibrated to UIUC low-Re database; 0.041 at Re=10k → 0.015 at Re=150k
// previous formula (0.044·Re⁻⁰·²²) floored at 0.005 for all Re > 19k → Cd0 constant → FM flat

F = (2/π)·arccos( exp(−f) ) f = (Nb/2)·(R−r) / (r·sin φ)
// Prandtl tip loss factor: F→0 at blade tip, F→1 inboard
// φ = atan(vi/Ut) decreases with RPM → F increases with RPM → FM rises

Ct = T / (ρ·A·(ΩR)²) Cp = P / (ρ·A·(ΩR)³)
// non-dimensional thrust and power coefficients (standard rotor form)

FM = Ct^(3/2) / (Cp · √2)
// momentum theory figure of merit — equivalent to P_ideal/P_actual
